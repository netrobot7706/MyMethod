## 修改后的 `sysex_decoder.c`（完整文件）

```c
#include "sysex_decoder.h"
#include <string.h>

// 弱符号默认实现（用户可覆盖）
SYSEX_WEAK void sysex_tx_aborted_notify(void) {
}

/* 统一熔断出口：优先使用函数指针，无则回退弱符号 */
static void abort_tx(sysex_decoder_t *d) {
    if (d->abort_handler != NULL) {
        d->abort_handler();
    } else {
        sysex_tx_aborted_notify();
    }
    d->in_multi_packet = false;
    d->recon_idx = 0u;
    d->expect_seg = 0u;
}

sysex_error_t sysex_decoder_init(sysex_decoder_t *decoder, uint8_t sub_id,
                                 sysex_json_handler_t json_handler,
                                 sysex_abort_handler_t abort_handler) {
    if (decoder == NULL) {
        return SYSEX_ERR_NULL_POINTER;
    }
    if (json_handler == NULL) {
        return SYSEX_ERR_NULL_POINTER;
    }

    decoder->sub_id = sub_id & 0x7Fu;
    decoder->json_handler = json_handler;
    decoder->abort_handler = abort_handler;
    decoder->expect_seg = 0u;
    decoder->recon_idx = 0u;
    decoder->in_multi_packet = false;
    decoder->initialized = true;
    (void)memset(decoder->recon_buf, 0, sizeof(decoder->recon_buf));
    return SYSEX_OK;
}

void sysex_decoder_reset(sysex_decoder_t *decoder) {
    if (decoder == NULL || !decoder->initialized) {
        return;
    }
    if (decoder->in_multi_packet) {
        abort_tx(decoder);
    } else {
        decoder->recon_idx = 0u;
        decoder->expect_seg = 0u;
    }
}

void sysex_decoder_process(sysex_decoder_t *decoder,
                           const uint8_t *data, size_t len) {
    if (decoder == NULL || !decoder->initialized || data == NULL) {
        return;
    }

    if (len < 2u || data[0] != 0xF0u || data[len-1u] != 0xF7u) {
        return;
    }

    const uint8_t *inner = &data[1];
    size_t inner_len = len - 2u;

    // 最小头部 + XOR：6 字节
    if (inner_len < 6u) {
        return;
    }

    // 1. 厂商 ID
    if (inner[0] != SYSEX_MANUFACTURER_ID) {
        return;
    }

    // 2. Sub-ID
    uint8_t sub_id = inner[1];
    if (!sysex_is_data_byte(sub_id) || sub_id != decoder->sub_id) {
        return;
    }

    // 3. Ver/Seg
    uint8_t ver_seg = inner[2];
    if (!sysex_is_data_byte(ver_seg)) {
        return;
    }
    uint8_t version = (uint8_t)((ver_seg >> 4) & 0x0Fu);
    if (version != SYSEX_PROTOCOL_VERSION) {
        return;
    }
    uint8_t segment = ver_seg & 0x0Fu;

    // 4. 长度字段
    uint8_t len_high = inner[3];
    uint8_t len_low  = inner[4];
    if (!sysex_is_data_byte(len_high) || !sysex_is_data_byte(len_low)) {
        return;
    }
    size_t declared_len = ((size_t)len_high << 7) | (size_t)len_low;

    // 帧完整性检查（用减法防止回绕）
    if (declared_len + 1u > inner_len - 5u) {
        return;
    }

    const uint8_t *payload = &inner[5];
    size_t payload_len = declared_len;

    // 检查 payload 全部为 7-bit
    for (size_t i = 0; i < payload_len; i++) {
        if (!sysex_is_data_byte(payload[i])) {
            return;
        }
    }

    // 5. XOR 校验
    uint8_t xor_calc = sub_id ^ ver_seg ^ len_high ^ len_low;
    for (size_t i = 0; i < payload_len; i++) {
        xor_calc ^= payload[i];
    }
    xor_calc &= 0x7Fu;
    uint8_t xor_recv = inner[5u + payload_len];
    if (!sysex_is_data_byte(xor_recv) || xor_calc != xor_recv) {
        return;
    }

    // 6. 单包超长检查（修复：仅多包事务才触发 abort）
    if (payload_len > (size_t)MAX_JSON_LEN) {
        if (decoder->in_multi_packet) {
            abort_tx(decoder);
        }
        return;
    }

    // ━━━ 状态机处理 ━━━
    if (segment == 0x1u) {
        // 首包
        if (decoder->in_multi_packet) {
            abort_tx(decoder);
        }
        decoder->recon_idx = 0u;
        decoder->expect_seg = 0x2u;
        decoder->in_multi_packet = true;
        (void)memcpy(decoder->recon_buf, payload, payload_len);
        decoder->recon_idx = payload_len;
    }
    else if (segment == 0x0u) {
        // 末包 / 单包
        if (decoder->in_multi_packet) {
            // 防御：多包事务中末包到来时，expect_seg 至少应为 0x02
            if (decoder->expect_seg < 0x02u) {
                abort_tx(decoder);
                return;
            }

            if (payload_len > ((size_t)MAX_JSON_LEN - decoder->recon_idx)) {
                abort_tx(decoder);
                return;
            }
            (void)memcpy(&decoder->recon_buf[decoder->recon_idx], payload, payload_len);
            decoder->recon_idx += payload_len;
            decoder->recon_buf[decoder->recon_idx] = '\0';
            decoder->json_handler(decoder->recon_buf, decoder->recon_idx);
        }
        else if (decoder->recon_idx == 0u) {
            // 纯单包
            (void)memcpy(decoder->recon_buf, payload, payload_len);
            decoder->recon_buf[payload_len] = '\0';
            decoder->json_handler(decoder->recon_buf, payload_len);
        }
        else {
            // 孤立末包
            abort_tx(decoder);
            return;
        }

        // 事务正常结束，复位
        decoder->recon_idx = 0u;
        decoder->expect_seg = 0u;
        decoder->in_multi_packet = false;
    }
    else {
        // 中间包 (0x2 ~ 0xF)
        if (decoder->in_multi_packet &&
            decoder->expect_seg <= 0x0Fu &&
            segment == decoder->expect_seg &&
            (payload_len <= ((size_t)MAX_JSON_LEN - decoder->recon_idx)))
        {
            (void)memcpy(&decoder->recon_buf[decoder->recon_idx], payload, payload_len);
            decoder->recon_idx += payload_len;
            decoder->expect_seg++;
        }
        else {
            abort_tx(decoder);
        }
    }
}
```
