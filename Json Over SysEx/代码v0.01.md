## 1. 公共定义 (`sysex_codec.h`)

```c
#ifndef SYSEX_CODEC_H
#define SYSEX_CODEC_H

#include <stdint.h>
#include <stddef.h>
#include <stdbool.h>

#ifdef __cplusplus
extern "C" {
#endif

#define SYSEX_MANUFACTURER_ID  0x7D
#define SYSEX_PROTOCOL_VERSION 0x1
#define SYSEX_MAX_PACKETS      15
#define SYSEX_MAX_PAYLOAD_PER_PACKET 256
#define SYSEX_HEADER_SIZE      6   // F0, 7D, SubID, VerSeg, LenH, LenL
#define SYSEX_FOOTER_SIZE      2   // XOR, F7

// 错误码（与协议一致）
typedef enum {
    SYSEX_OK = 0,
    SYSEX_ERR_INVALID_FRAME = -1,
    SYSEX_ERR_WRONG_MANUFACTURER = -2,
    SYSEX_ERR_UNSUPPORTED_VERSION = -3,
    SYSEX_ERR_LENGTH_MISMATCH = -4,
    SYSEX_ERR_CHECKSUM_FAIL = -5,
    SYSEX_ERR_BUFFER_TOO_SMALL = -6,
    SYSEX_ERR_TOO_LARGE = -7,        // 总长度超出最大15包限制
} sysex_error_t;

#ifdef __cplusplus
}
#endif

#endif
```

## 2. 编码器 (`sysex_encoder.h` / `.c`)

### `sysex_encoder.h`

```c
#ifndef SYSEX_ENCODER_H
#define SYSEX_ENCODER_H

#include "sysex_codec.h"

typedef struct {
    const uint8_t *json_data;   // 输入 JSON 数据指针
    size_t         json_len;    // 输入长度
    uint8_t        sub_id;      // 设备子标识
    uint8_t        total_packets;
    uint8_t        current_packet;
} sysex_encoder_t;

/**
 * 初始化编码器上下文
 * @param ctx       编码器上下文
 * @param json      JSON 字符串（必须为 ASCII，即所有字节 <= 0x7F）
 * @param len       JSON 字节长度（不含 '\0'）
 * @param sub_id    设备子标识 (0x00-0x7F)
 * @return SYSEX_OK 成功，SYSEX_ERR_TOO_LARGE 数据过长
 */
sysex_error_t sysex_encoder_init(sysex_encoder_t *ctx,
                                 const uint8_t *json, size_t len,
                                 uint8_t sub_id);

/**
 * 获取总包数（仅在初始化后有效）
 */
static inline uint8_t sysex_encoder_packet_count(const sysex_encoder_t *ctx) {
    return ctx->total_packets;
}

/**
 * 获取下一帧数据
 * @param ctx       编码器上下文
 * @param out_buf   输出缓冲区，需至少能容纳 8 + 256 = 264 字节
 * @param out_len   实际输出的帧长度（包括 F0/F7）
 * @return true  成功填充一帧
 * @return false 所有帧已发送完毕
 */
bool sysex_encoder_next(sysex_encoder_t *ctx,
                        uint8_t *out_buf, size_t *out_len);

#endif
```

### `sysex_encoder.c`

```c
#include "sysex_encoder.h"
#include <string.h>

sysex_error_t sysex_encoder_init(sysex_encoder_t *ctx,
                                 const uint8_t *json, size_t len,
                                 uint8_t sub_id) {
    if (len > SYSEX_MAX_PACKETS * SYSEX_MAX_PAYLOAD_PER_PACKET) {
        return SYSEX_ERR_TOO_LARGE;
    }
    ctx->json_data = json;
    ctx->json_len  = len;
    ctx->sub_id    = sub_id & 0x7F;
    ctx->total_packets = (uint8_t)((len + SYSEX_MAX_PAYLOAD_PER_PACKET - 1) /
                                   SYSEX_MAX_PAYLOAD_PER_PACKET);
    if (ctx->total_packets == 0) ctx->total_packets = 1;
    ctx->current_packet = 0;
    return SYSEX_OK;
}

bool sysex_encoder_next(sysex_encoder_t *ctx,
                        uint8_t *out_buf, size_t *out_len) {
    if (ctx->current_packet >= ctx->total_packets) {
        return false;
    }

    uint8_t idx = ctx->current_packet;
    size_t offset = idx * SYSEX_MAX_PAYLOAD_PER_PACKET;
    size_t chunk_len = ctx->json_len - offset;
    if (chunk_len > SYSEX_MAX_PAYLOAD_PER_PACKET) {
        chunk_len = SYSEX_MAX_PAYLOAD_PER_PACKET;
    }

    // 确定 segment
    uint8_t seg;
    if (ctx->total_packets == 1) {
        seg = 0x0;
    } else if (idx == 0) {
        seg = 0x1;
    } else if (idx == ctx->total_packets - 1) {
        seg = 0x0;
    } else {
        seg = 0x2 + (idx - 1);
    }

    uint8_t ver_seg = (SYSEX_PROTOCOL_VERSION << 4) | (seg & 0x0F);
    uint8_t len_high = (chunk_len >> 7) & 0x7F;
    uint8_t len_low  = chunk_len & 0x7F;

    // 计算 XOR
    uint8_t xor_val = ctx->sub_id ^ ver_seg ^ len_high ^ len_low;
    for (size_t i = 0; i < chunk_len; i++) {
        xor_val ^= ctx->json_data[offset + i];
    }
    xor_val &= 0x7F;

    // 构建帧
    size_t pos = 0;
    out_buf[pos++] = 0xF0;
    out_buf[pos++] = SYSEX_MANUFACTURER_ID;
    out_buf[pos++] = ctx->sub_id;
    out_buf[pos++] = ver_seg;
    out_buf[pos++] = len_high;
    out_buf[pos++] = len_low;
    memcpy(&out_buf[pos], ctx->json_data + offset, chunk_len);
    pos += chunk_len;
    out_buf[pos++] = xor_val;
    out_buf[pos++] = 0xF7;
    *out_len = pos;

    ctx->current_packet++;
    return true;
}
```

## 3. 解码器 (`sysex_decoder.h` / `.c`)

严格遵循 V1.2.1-Final 状态机，含 `MAX_JSON_LEN` 硬限制、abort 通知回调。

### `sysex_decoder.h`

```c
#ifndef SYSEX_DECODER_H
#define SYSEX_DECODER_H

#include "sysex_codec.h"

// 用户必须在包含此头文件前定义 MAX_JSON_LEN
#ifndef MAX_JSON_LEN
#error "MAX_JSON_LEN must be defined before including sysex_decoder.h"
#endif

// 业务层回调
typedef void (*sysex_json_handler_t)(const char *json, size_t len);
typedef void (*sysex_abort_handler_t)(void);

typedef struct {
    uint8_t  sub_id;
    uint8_t  expect_seg;
    size_t   recon_idx;
    bool     in_multi_packet;
    uint8_t  recon_buf[MAX_JSON_LEN + 1];  // +1 for null terminator

    sysex_json_handler_t  json_handler;
    sysex_abort_handler_t abort_handler;
} sysex_decoder_t;

/**
 * 初始化解码器
 * @param decoder        解码器上下文
 * @param sub_id         期望的设备子标识
 * @param json_handler   重组完成后的 JSON 回调
 * @param abort_handler  多包事务异常中断时的回调（可为 NULL）
 */
void sysex_decoder_init(sysex_decoder_t *decoder, uint8_t sub_id,
                        sysex_json_handler_t json_handler,
                        sysex_abort_handler_t abort_handler);

/**
 * 处理一帧完整的 SysEx 消息（含 F0/F7）
 * @param decoder  解码器上下文
 * @param data     SysEx 消息字节数组
 * @param len      数组长度
 */
void sysex_decoder_process(sysex_decoder_t *decoder,
                           const uint8_t *data, size_t len);

/**
 * 手动重置状态机（外部故障恢复）
 */
void sysex_decoder_reset(sysex_decoder_t *decoder);

#endif
```

### `sysex_decoder.c`

```c
#include "sysex_decoder.h"
#include <string.h>

// 弱符号实现 abort 通知，可在用户侧覆盖
__attribute__((weak)) void sysex_tx_aborted_notify(void) {
    // 默认空实现
}

static void abort_tx(sysex_decoder_t *d) {
    if (d->in_multi_packet) {
        if (d->abort_handler) {
            d->abort_handler();
        }
        d->in_multi_packet = false;
    }
    d->recon_idx = 0;
    d->expect_seg = 0;
}

void sysex_decoder_init(sysex_decoder_t *decoder, uint8_t sub_id,
                        sysex_json_handler_t json_handler,
                        sysex_abort_handler_t abort_handler) {
    decoder->sub_id = sub_id & 0x7F;
    decoder->json_handler = json_handler;
    decoder->abort_handler = abort_handler;
    decoder->expect_seg = 0;
    decoder->recon_idx = 0;
    decoder->in_multi_packet = false;
    memset(decoder->recon_buf, 0, sizeof(decoder->recon_buf));
}

void sysex_decoder_reset(sysex_decoder_t *decoder) {
    if (decoder->in_multi_packet && decoder->abort_handler) {
        decoder->abort_handler();
    }
    decoder->expect_seg = 0;
    decoder->recon_idx = 0;
    decoder->in_multi_packet = false;
}

void sysex_decoder_process(sysex_decoder_t *decoder,
                           const uint8_t *data, size_t len) {
    if (len < 2 || data[0] != 0xF0 || data[len-1] != 0xF7) {
        return; // 无效帧静默丢弃
    }

    const uint8_t *inner = data + 1;  // 跳过 F0
    size_t inner_len = len - 2;       // 去掉 F0/F7

    if (inner_len < 5) return;  // 最小头部

    // 1. 厂商 ID
    if (inner[0] != SYSEX_MANUFACTURER_ID) return;

    // 2. Sub-ID
    uint8_t sub_id = inner[1];
    if (sub_id != decoder->sub_id) return; // 非本设备

    // 3. Ver/Seg
    uint8_t ver_seg = inner[2];
    uint8_t version = (ver_seg >> 4) & 0x0F;
    if (version != SYSEX_PROTOCOL_VERSION) return;

    uint8_t segment = ver_seg & 0x0F;

    // 4. 长度
    uint8_t len_high = inner[3];
    uint8_t len_low  = inner[4];
    size_t declared_len = ((size_t)len_high << 7) | len_low;

    // 检查帧内实际 payload 大小
    if (inner_len < 5 + declared_len + 1) return; // +1 for XOR

    const uint8_t *payload = &inner[5];
    size_t payload_len = declared_len;

    // 5. XOR 校验
    uint8_t xor_calc = sub_id ^ ver_seg ^ len_high ^ len_low;
    for (size_t i = 0; i < payload_len; i++) {
        xor_calc ^= payload[i];
    }
    xor_calc &= 0x7F;
    uint8_t xor_recv = inner[5 + payload_len];
    if (xor_calc != xor_recv) return;

    // 6. 检查 payload 字节 ≤ 0x7F
    for (size_t i = 0; i < payload_len; i++) {
        if (payload[i] > 0x7F) return;
    }

    // ━━━ 状态机处理 ━━━
    if (payload_len > MAX_JSON_LEN) {
        abort_tx(decoder);
        return;
    }

    if (segment == 0x1) {
        // 首包
        if (decoder->in_multi_packet) {
            abort_tx(decoder); // 上一事务异常中断通知
        }
        decoder->recon_idx = 0;
        decoder->expect_seg = 0x2;
        decoder->in_multi_packet = true;
        memcpy(decoder->recon_buf, payload, payload_len);
        decoder->recon_idx = payload_len;
    }
    else if (segment == 0x0) {
        // 末包 / 单包
        if (decoder->in_multi_packet) {
            if (decoder->recon_idx + payload_len > MAX_JSON_LEN) {
                abort_tx(decoder);
                return;
            }
            memcpy(decoder->recon_buf + decoder->recon_idx, payload, payload_len);
            decoder->recon_idx += payload_len;
            decoder->recon_buf[decoder->recon_idx] = '\0';
            if (decoder->json_handler) {
                decoder->json_handler((const char*)decoder->recon_buf, decoder->recon_idx);
            }
        }
        else if (decoder->recon_idx == 0) {
            // 纯单包
            memcpy(decoder->recon_buf, payload, payload_len);
            decoder->recon_buf[payload_len] = '\0';
            if (decoder->json_handler) {
                decoder->json_handler((const char*)decoder->recon_buf, payload_len);
            }
        }
        else {
            abort_tx(decoder);
            return;
        }
        // 正常复位
        decoder->recon_idx = 0;
        decoder->expect_seg = 0;
        decoder->in_multi_packet = false;
    }
    else {
        // 中间包 0x2 - 0xF
        if (decoder->in_multi_packet &&
            segment == decoder->expect_seg &&
            (decoder->recon_idx + payload_len <= MAX_JSON_LEN))
        {
            memcpy(decoder->recon_buf + decoder->recon_idx, payload, payload_len);
            decoder->recon_idx += payload_len;
            decoder->expect_seg++;
        }
        else {
            abort_tx(decoder);
        }
    }
}
```
